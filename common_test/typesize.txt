## BASIC Types

[char] is	1 Byte.
[char*] is	8 Byte.
CHAR_min =	-128
CHAR_MAX =	127

[short] is	2 Byte.
[short*] is	8 Byte.
SHRT_min =	-32768
, SHRT_MAX =	32767

[int] is	4 Byte.
[int*] is	8 Byte.
INT_min =	-2147483648, INT_MAX =	2147483647

[long] is	8 Byte.
[long*] is	 8 Byte.
LONG_min =	-9223372036854775808, LONG_MAX =	9223372036854775807

[long long] is	8 Byte.
[long long*] is	8 Byte.

[float] is	4 Byte.
[float*] is	8 Byte.

[double] is	8 Byte.
[double*] is	8 Byte.

[long double] is	16 Byte.
[long double*] is	8 Byte.
